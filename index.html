Custom Data Types


            puts "\nCustom Data Types #{sectionLine}"
            class Person
                def initialize(name, age)
                    @name = name
                    self.age = age
                end
            
                # Get name
                def name
                    @name
                end
            
                # Get age
                def age
                    @age
                end
            
                def name=(new_name)
                    @name = new_name
                end
            
                def age=(new_age)
                    if new_age >= 0
                        @age = new_age
                    else
                        puts "Trying age #{new_age}. Age cannot be negative!"
                    end
                end
            
                def introduce
                    "Hello, I'm #{@name}, and I'm #{@age} years old!"
                end
            end
            
            puts "Creating Person..."
            person = Person.new("Jack", -5)
            puts "Changing Age..."
            person.age = 24
            puts person.introduce
